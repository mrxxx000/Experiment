def verify_token(token: str) -> dict:
    if token == "valid_manager_token":
        return {"sub": "manager_user", "role": "manager"}
    elif token == "valid_employee_token":
        return {"sub": "regular_user", "role": "employee"}
    else:
        raise HTTPException(
            status_code=status.HTTP_401_UNAUTHORIZED,
            detail="Invalid or expired token",
            headers={"WWW-Authenticate": "Bearer"},
        )


def require_role(role: str):
    def role_checker(credentials: HTTPAuthorizationCredentials = Depends(security)) -> dict:
        token = credentials.credentials
        payload = verify_token(token)
        if payload.get("role") != role:
            raise HTTPException(
                status_code=status.HTTP_403_FORBIDDEN,
                detail=f"Access denied: '{role}' role required",
            )
        return payload
    return role_checker


@app.get("/employees", response_model=List[Employee])
def get_employees(
    current_user: dict = Depends(require_role("manager"))
) -> List[Employee]:
