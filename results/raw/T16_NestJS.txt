import {
  Controller,
  ForbiddenException,
  Get,
  NotFoundException,
  Param,
  Req,
  UseGuards,
} from '@nestjs/common';
import { AuthGuard } from '@nestjs/passport';
import { Request } from 'express';
import { DocumentService } from './document.service';
import { Document } from './document.entity';

type AuthenticatedRequest = Request & {
  user: {
    id: string;
  };
};

@Controller('documents')
export class DocumentController {
  constructor(private readonly documentService: DocumentService) {}

  @Get(':documentId')
  @UseGuards(AuthGuard('jwt'))
  async getDocument(
    @Param('documentId') documentId: string,
    @Req() req: AuthenticatedRequest,
  ): Promise<Document> {
    const document = await this.documentService.findById(documentId);

    if (!document) {
      throw new NotFoundException('Document not found');
    }

    if ((document as Document & { userId: string }).userId !== req.user.id) {
      throw new ForbiddenException('You do not have access to this document');
    }

    return document;
  }
}

import {
  Controller,
  ForbiddenException,
  Get,
  NotFoundException,
  Param,
  Req,
  UseGuards,
} from '@nestjs/common';
import { AuthGuard } from '@nestjs/passport';
import { Request } from 'express';
import { DocumentService } from './document.service';
import { Document } from './document.entity';

type AuthenticatedRequest = Request & {
  user: {
    id?: string;
    userId?: string;
    sub?: string;
  };
};

@Controller('documents')
export class DocumentController {
  constructor(private readonly documentService: DocumentService) {}

  @UseGuards(AuthGuard('jwt'))
  @Get(':documentId')
  async findById(
    @Param('documentId') documentId: string,
    @Req() req: AuthenticatedRequest,
  ): Promise<Document> {
    const document = await this.documentService.findById(documentId);

    if (!document) {
      throw new NotFoundException('Document not found');
    }

    const authenticatedUserId = req.user.id ?? req.user.userId ?? req.user.sub;

    const documentOwnerId =
      (document as any).userId ??
      (document as any).ownerId ??
      (document as any).user?.id ??
      (document as any).owner?.id;

    if (!authenticatedUserId || String(documentOwnerId) !== String(authenticatedUserId)) {
      throw new ForbiddenException();
    }

    return document;
  }
}
