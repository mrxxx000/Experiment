export const REQUIRE_SUBSCRIPTION = 'require_subscription';
export const RequireSubscription = (plan: string) =>
  SetMetadata(REQUIRE_SUBSCRIPTION, plan);

@Injectable()
export class SubscriptionGuard implements CanActivate {
  constructor(private readonly reflector: Reflector) {}

  canActivate(context: ExecutionContext): boolean {
    const requiredPlan = this.reflector.getAllAndOverride<string>(
      REQUIRE_SUBSCRIPTION,
      [context.getHandler(), context.getClass()],
    );

    if (!requiredPlan) {
      return true;
    }

    const request = context.switchToHttp().getRequest<AuthenticatedRequest>();
    const user = request.user;

    if (!user) {
      throw new UnauthorizedException('Authentication required');
    }

    const subscription = user.subscription;

    if (
      !subscription ||
      subscription.plan !== requiredPlan ||
      subscription.status !== 'active'

      throw new ForbiddenException(
        `An active "${requiredPlan}" subscription is required to access this resource`,

    return true;
  }
