## database.py — In-Memory User Store
```python
"""
Lightweight in-memory store that mimics a real database.
Replace with SQLAlchemy / async DB calls in production.
"""
from typing import Dict, Optional
from models import UserInDB


class FakeDatabase:
    def __init__(self) -> None:
        self._users: Dict[str, UserInDB] = {}
        self._id_counter: int = 1

    # ── Write ────────────────────────────────────────────────────────────────
    def create_user(self, user: UserInDB) -> UserInDB:
        if user.username in self._users:
            raise ValueError(f"Username '{user.username}' already exists.")
        user.id = self._id_counter
        self._id_counter += 1
        self._users[user.username] = user
        return user

    # ── Read ─────────────────────────────────────────────────────────────────
    def get_by_username(self, username: str) -> Optional[UserInDB]:
        return self._users.get(username)

    def get_by_id(self, user_id: int) -> Optional[UserInDB]:
        return next(
            (u for u in self._users.values() if u.id == user_id), None
        )

    def list_users(self) -> list[UserInDB]:
        return list(self._users.values())


# Singleton instance shared across the app
db = FakeDatabase()
```
